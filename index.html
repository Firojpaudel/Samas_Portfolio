class ContactManager:
    """Contact management system for portfolio"""
    
    def __init__(self):
        self.contact_info = {
            "email": "samas@example.com",
            "linkedin": "linkedin.com/in/samas",
            "github": "github.com/samas",
            "twitter": "@samas_dev",
            "instagram": "instagram.com/samas",
            "discord": "discord.com/users/yourdiscordid"
        }
    
    def display_contact_info(self) -> None:
        """Display all contact information"""
        print("Let's Connect! ü§ù")
        print("=" * 30)
        
        for platform, value in self.contact_info.items():
            print(f"{platform.title()}: {value}")
    
    def send_message(self, name: str, email: str, message: str) -> bool:
        """Simulate sending a message"""
        print(f"\nüìß Message from {name} ({email}):")
        print(f"'{message}'")
        print("Message sent successfully! üìß")
        return True
